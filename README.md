## Install

```bash
npm install --save bornfight
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'bornfight'
import 'bornfight/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
